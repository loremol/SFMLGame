cmake_minimum_required(VERSION 3.18)
project(SFMLGame)
set(CMAKE_CXX_STANDARD 17)
set(SFML_STATIC_LIBRARIES TRUE)
set(SOURCE_FILES main.cpp Map/Tile.cpp Map/Tile.hpp Map/World.cpp Map/World.hpp Entities/Entity.hpp Entities/Player.cpp Entities/Player.hpp Entities/Entity.cpp States/State.hpp States/StateMachine.cpp States/StateMachine.hpp AssetManager.cpp AssetManager.hpp InputManager.cpp InputManager.hpp Game.cpp Game.hpp Definitions.hpp States/SplashState.cpp States/SplashState.hpp States/MainMenuState.cpp States/MainMenuState.hpp States/GameState.cpp States/GameState.hpp States/Button.cpp States/Button.hpp States/PlayButton.cpp States/PlayButton.hpp States/ExitButton.cpp States/ExitButton.hpp Components/MovementComponent.cpp Components/MovementComponent.hpp)
add_executable(SFMLGame ${SOURCE_FILES})

if(WIN32)
    set(SFML_DIR C:/SFML/lib/cmake/SFML)
else()
    set(SFML_DIR /usr/lib/x86_64-linux-gnu/cmake/SFML)
endif ()

find_package(SFML COMPONENTS graphics REQUIRED)
if(SFML_FOUND)
    include_directories(${SFML_INCLUDE_DIR})
    target_link_libraries(SFMLGame sfml-graphics)
endif()
